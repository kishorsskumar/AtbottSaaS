Create a new folder "backend/projects" and perform the following steps.

1. Create the file "backend/projects/manager.py" with this code:

---
import os, json, shutil
from fastapi import APIRouter, Body, HTTPException

router = APIRouter()
PROJECTS_DIR = os.path.abspath("projects")
os.makedirs(PROJECTS_DIR, exist_ok=True)

def project_path(name):
    return os.path.join(PROJECTS_DIR, name)

@router.get("/")
def list_projects():
    return {"projects": os.listdir(PROJECTS_DIR)}

@router.post("/create")
def create_project(name: str = Body(..., embed=True)):
    path = project_path(name)
    if os.path.exists(path):
        raise HTTPException(400, f"Project {name} already exists")
    os.makedirs(path)
    with open(os.path.join(path, "README.md"), "w") as f:
        f.write(f"# {name}\nGenerated by Atbott SaaS Project\n")
    return {"created": name}

@router.delete("/delete")
def delete_project(name: str = Body(..., embed=True)):
    path = project_path(name)
    if not os.path.exists(path):
        raise HTTPException(404, "Project not found")
    shutil.rmtree(path)
    return {"deleted": name}

@router.get("/open")
def open_project(name: str):
    path = project_path(name)
    if not os.path.exists(path):
        raise HTTPException(404, "Project not found")
    files = []
    for root, _, file_list in os.walk(path):
        for f in file_list:
            full = os.path.join(root, f)
            files.append(full.replace(PROJECTS_DIR + os.sep, ""))
    return {"project": name, "files": files}
---

2. In backend/main.py → add:
---
from backend.projects import manager
app.include_router(manager.router, prefix="/projects", tags=["Project Manager"])
---

3. Restart backend:
   python backend/main.py

4. Open /docs → confirm you see "Project Manager" with routes:
   - GET /projects/  
   - POST /projects/create  
   - DELETE /projects/delete  
   - GET /projects/open  

These give you basic multi‑project control.

5. Frontend UI:
In frontend/src, create "ProjectsDashboard.jsx" with:

---
import { useEffect, useState } from "react";
import axios from "axios";

export default function ProjectsDashboard({ onOpen }) {
  const [projects, setProjects] = useState([]);
  const [name, setName] = useState("");
  const BASE = "[atbott-saas-project](https://atbott-saas-project)<YOUR_USERNAME>.repl.co/projects";

  const refresh = async () => {
    const res = await axios.get(BASE + "/");
    setProjects(res.data.projects);
  };
  useEffect(() => { refresh(); }, []);

  const create = async () => {
    await axios.post(BASE + "/create", { name });
    setName("");
    refresh();
  };
  const del = async (n) => {
    await axios.delete(BASE + "/delete", { data: { name: n }});
    refresh();
  };

  return (
    <div style={{padding:20, background:"#181818", color:"#fff", height:"100vh"}}>
      <h2>Atbott SaaS Project Launcher</h2>
      <input value={name} onChange={e=>setName(e.target.value)} placeholder="New project name" />
      <button onClick={create}>➕ Create</button>
      <ul>
        {projects.map(p=>(
          <li key={p}>
            <span onClick={()=>onOpen(p)} style={{cursor:"pointer"}}>🚀 {p}</span>
            <button onClick={()=>del(p)}>🗑️</button>
          </li>
        ))}
      </ul>
    </div>
  );
}
---

6. In frontend/src/main.jsx replace its root renderer with:

---
import React from 'react'
import ReactDOM from 'react-dom/client'
import ProjectsDashboard from './ProjectsDashboard'
import FileManager from './FileManager'
import './index.css'

function App() {
  const [project, setProject] = React.useState("")
  return project ? <FileManager basePath={`projects/${project}`} /> : <ProjectsDashboard onOpen={setProject}/>
}

ReactDOM.createRoot(document.getElementById('root')).render(
  <React.StrictMode><App/></React.StrictMode>,
)
---

7. Run the frontend:
   cd frontend  
   npm run dev

8. You now have a Replit‑style dashboard:
   - Create new projects  
   - Open any project → loads its files in the editor panel  
   - Delete projects not needed  

Once you see this working, display:
"✅ Phase 9 workspace platform functional."
