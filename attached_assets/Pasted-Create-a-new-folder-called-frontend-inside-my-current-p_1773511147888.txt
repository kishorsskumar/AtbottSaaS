Create a new folder called "frontend" inside my current project.

In that folder, create a new Vite + React app configured for JavaScript (not TypeScript).

Then perform these exact steps:

1. Inside the new "frontend" folder, run:
   npm init vite@latest . -- --template react
   npm install
   npm install axios @monaco-editor/react

2. In "frontend/package.json", add this script so it runs on Replit:
   "scripts": {
      "dev": "vite --port 5173 --host"
   }

3. In "frontend/src/App.jsx", replace everything with:

---
import { useState } from 'react';
import Editor from '@monaco-editor/react';
import axios from 'axios';

export default function App() {
  const [prompt, setPrompt] = useState('');
  const [output, setOutput] = useState('');
  const [loading, setLoading] = useState(false);

  const sendPrompt = async () => {
    if (!prompt.trim()) return;
    setLoading(true);
    try {
      const res = await axios.post(
        '[atbott-saas-project](https://atbott-saas-project)<YOUR_USERNAME>.repl.co/generate',
        { message: prompt }
      );
      const text = res.data.content?.[0]?.text || JSON.stringify(res.data);
      setOutput(text);
    } catch (err) {
      setOutput('Error: ' + err.message);
    }
    setLoading(false);
  };

  return (
    <div style={{
      display:'flex',
      flexDirection:'column',
      height:'100vh',
      padding:'20px',
      background:'#1e1e1e',
      color:'#fff'
    }}>
      <h2>Atbott SaaS Project Workspace</h2>
      <textarea
        placeholder="Describe what you want to build or generate…"
        style={{flex:1, padding:10, fontSize:'16px', marginBottom:10}}
        onChange={e => setPrompt(e.target.value)}
        value={prompt}
      />
      <button
        style={{padding:10, background:'#28a', color:'#fff', border:'none', borderRadius:4}}
        onClick={sendPrompt}
      >
        {loading ? 'Generating…' : 'Run Prompt'}
      </button>
      <div style={{
        flex:2,
        background:'#111',
        marginTop:20,
        overflow:'auto',
        padding:10,
        borderRadius:4
      }}>
        <pre>{output}</pre>
      </div>
    </div>
  );
}
---

4. Replace <YOUR_USERNAME> in the axios URL with your actual Replit username.

5. In the Replit shell, run this to start the frontend:
   cd frontend
   npm run dev

6. After it builds, click **Open in New Tab** → you’ll see your Atbott SaaS Project Workspace UI.
